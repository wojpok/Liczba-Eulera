Algorytm regresji liniowej¶

Zosta zaimplementowany zgodnie ze przeksztaceniami zawartymi we sprawozdaniu. Funkcja liniowej regresji posikuje si fukcj pomocnicz $mat()$ ktra tworzy macierze o zadanych rozmiarach.

# Konstruktor macierzy X*Y o wartoci domylnej initialV
function mat(dimX, dimY, initialV)
    mat1 = Array{__precision}(undef, dimX, dimY)
    
    for i in 1:dimX
        for j in 1:dimY
            mat1[i, j] = initialV
        end
    end
    
    return mat1
end

# Liniowa regresja
function linear_regression(argumentsX, valuesY, verbose = false) # -> ax + b
    len = length(argumentsX)
	
	matX = mat(2, len, 1)
	matY = mat(1, len, 1)
	
	for i in 1:len
		matX[2, i] = argumentsX[i]
		matY[1, i] = valuesY[i]
	end

	a = (matX*transpose(matX))^-1* (matX) *transpose( matY)
	
    # rozszrzony output, podaje informacje o dokadnoci przyblienia dla kadego punktu
    if verbose
        @printf("Przyblienie a: %f\nb: %f\n", a[2,1], a[1,1])
        
        @printf("Lp. / X / Y / przyblienie / bd wzgldny / bd bezwgldny")
        
        minBW, maxBW = 10, 0
        minBB, maxBB = 10, 0
        
        for i in 1:len
            g = a[2,1]*matX[2, i] + a[1,1]
            
            bw = abs(valuesY[i] - g)
            bb = abs((valuesY[i] - g)/ valuesY[i])
            
            minBW = bw < minBW ? bw : minBW
            maxBW = bw > maxBW ? bw : maxBW
            
            minBB = bb < minBB ? bb : minBB
            maxBB = bb > maxBB ? bb : maxBB
            
            @printf("%d: %.10f %.10f %.10f %.10f %.10f\n",i, argumentsX[i], valuesY[i], g, bw, bb)
        end
        
        @printf("maxBW: %.10f, minBW: %.10f \nmaxBB: %.10f, minBB: %.10f\n", maxBW, minBW, maxBB, minBB)
    end
    return a[2,1], a[1,1] 
end

# stablicowana warto staej Eulera
eulerConstant = 0.57721566490153286060651209008240243104215933593992

# n-ty wyraz cigu gamma
function gamma(n)
    sum = zero(__precision)

    for i in 1:n
        sum = sum + 1/convert(__precision, (n - i + 1))
    end

    return  sum - log(convert(__precision, n))
end

# bd przyblienia dla n - przydatne rwnie dla drugiej metody
function error(n)
    return gamma(n) - eulerConstant
end

# konstruktor zbiorw wartoci; zwraca list elementw midzu begX i endX oddalonych od siebie o warto step
function inputgen(begX, endX, step)
    res = []
    x = begX
    while(x < endX)
        push!(res, x)
        x = x + step
    end
    
    return res
end


calcC = 0
calcD = 0

# waciwa fukncja obliczajca c oraz d
function compute(inputs; verbose = false)
    valuesY = []
    argumentsX = []
    chartY = []
    
    # inicjacja tablic z wartociami
    for i in 1:length(inputs)
        push!(chartY, error(inputs[i]))
        push!(valuesY, log2(chartY[i]))
        push!(argumentsX, log2(inputs[i]))
    end
    
    a, b = linear_regression(argumentsX, valuesY, verbose)
    
    c = 2^b
    d = -a
    
    # eby mc zwraca C oraz D i jednoczenie wywietla wykres jako wynik ewaluacji
    # wartoci s zwracane jako zmienne globalne
    global calcC, calcD
    
    calcC = c
    calcD = d
    
    # rozszerzony output funkcji. Podaje wyliczone c oraz d i przedstawia porwnanie cigu i funkcji na wykresie
    if verbose
        f(x) = c*x^(-d)

        @printf("c = %.20f\nd = %.20f\n", c, d)

        plot(LinRange(inputs[1], inputs[length(inputs)], 100), f , color="red", linewidth=2.0, #lineslista1/Untitled3tyle=:dash, 
        title="Porwnanie na przedziale ["*string(inputs[1])* "; " *string(inputs[length(inputs)])*"]", legend=:outertopleft, xlabel="\$x\$", label ="\$ cn^{-d} \$")

        Plots.scatter!(inputs, chartY , color="blue", label="\$ \\gamma_n - \\gamma \$" )
    end
end

# zaimplementowana metoda obliczania wykadnika zbieznoci
function convergenceExponent(n)
	return log2(error(n)/error(n-1))/log2(error(n-1)/error(n-2))
end

# druga metoda obliczeniowa, oblicza kolejne przyblienia d i zwraca wykres
function compute2(inputs; verbose = false)
    valuesY = []
    
    # inicjacja tablic z wartociami
    for i in 1:length(inputs)
        push!(valuesY, convergenceExponent(inputs[i]))
    end 
    
    
    # rozszerzony output funkcji
    if verbose
         @printf("Lp. / Wynik\n")
        for i in 1:length(inputs)
            @printf("%d %.10f\n", i, valuesY[i]);
        end
        
        
        Plots.scatter(inputs, valuesY , color="blue", title="Wykadnik zbieznoci", legend=:outertopleft, label ="\$ c \$")
        
        f(x) = 1
        
        plot!(LinRange(inputs[1], inputs[length(inputs)], 100), f , color="red", label="\$ f(x) = 1 \$" )
    end
end

# funkcja porwnujca obie metody, zaimplementowana jak we sprawozdaniu
function compare(inputs; verbose = false)
    valuesY1 = []
    valuesY2 = []
    
    for i in 1:length(inputs)
        compute([inputs[i]-2, inputs[i]-1, inputs[i]])
        push!(valuesY1, calcD)
        push!(valuesY2, convergenceExponent(inputs[i]))
        
       
    end 
    
    if verbose
         @printf("Lp. / Metoda 1 / Metoda 2\n")
        for i in 1:length(inputs)
            @printf("%d %.10f %.10f\n", i, valuesY1[i], valuesY2[i]);
        end
        
        Plots.scatter(inputs, valuesY1,  color="blue", title="Wykadnik zbieznoci", legend=:outertopleft, label ="Metoda 1")
        Plots.scatter!(inputs, valuesY2, color="yellow", label ="Metoda 2")

        f(x) = 1

        plot!(LinRange(inputs[1], inputs[length(inputs)], 100), f , color="red", label="\$ f(x) = 1 \$" )
    end
end
